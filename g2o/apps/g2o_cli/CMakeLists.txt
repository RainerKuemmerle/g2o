  dl_wrapper.cpp dl_wrapper.h
  output_helper.cpp output_helper.h
  g2o_common.cpp g2o_common.h
  g2o_cli_api.h
)

endif()

  target_link_libraries(g2o_cli_library ${CMAKE_DL_LIBS})
